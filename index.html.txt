<script>
async function generate() {
  const data = {
    name: name.value,
    role: role.value,
    skills: skills.value,
    experience: exp.value
  };

  const res = await fetch("/api/resume", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(data)
  });

  const result = await res.json();
  output.textContent = result.resume;
}
</script>
